# 相較於其他年齡層，由於高年齡層的客戶基數較少，導致轉換率的差異太過於明顯
# 將各個年齡層做合併，擴大基數來解決這個問題
bank_add_df["age_group"] = bank_add_df["age"].apply(
    lambda x: "18 to 29" if x < 30 else
              "30 to 39" if x < 40 else
              "40 to 49" if x < 50 else
              "50 to 59" if x < 60 else
              "60 to 69" if x < 70 else
              "70+")

# 將年齡層套入婚姻狀態
agegroup_marital = bank_add_df.groupby(["age_group", "marital"])[
    "conversion"].sum().unstack("marital").fillna(0)
agegroup_marital = agegroup_marital.divide(bank_add_df.groupby(
    by="age_group")["conversion"].count(), axis=0)
